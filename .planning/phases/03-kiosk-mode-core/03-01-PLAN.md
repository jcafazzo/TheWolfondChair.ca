```javascript
function enterKiosk() {
  isKioskMode = true;
  document.body.setAttribute('data-kiosk', '');
  startAutoAdvance(); // fresh timer from current slide
  // Request fullscreen (per user decision)
  if (document.documentElement.requestFullscreen) {
    document.documentElement.requestFullscreen().catch(() => {});
  }
  // Start cursor hide timer
  startCursorHideTimer();
}
```

**4. Add exitKiosk() function:**
```javascript
function exitKiosk() {
  isKioskMode = false;
  document.body.removeAttribute('data-kiosk');
  stopAutoAdvance();
  // Exit fullscreen if active
  if (document.fullscreenElement) {
    document.exitFullscreen().catch(() => {});
  }
  // Clear cursor hide timer, restore cursor
  clearCursorHideTimer();
  // Restart normal auto-advance
  startAutoAdvance();
}
```

**5. Add cursor hide helpers:**

```javascript
function startCursorHideTimer() {

function clearCursorHideTimer() {
  if (cursorHideTimer) { clearTimeout(cursorHideTimer); cursorHideTimer = null; }
  document.body.style.cursor = '';
}
```
